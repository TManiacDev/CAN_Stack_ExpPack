354 FUNC(Std_ReturnType, TM_ISOTP_CODE) IsoTp::ReceiveFlowControl( P2CONST(CanIsoTP_8ByteData, AUTOMATIC, AUTOMATIC) ptr2FlowControlFrame )
355 {
356  Std_ReturnType returnValue = E_NOT_OK;
357 
358  /* handle fc frame only when sending in progress */
359  if ( objectHandle.FlowControl.ProtocolStatus == IsoTP_RxTxStatus::IsoTP_SendInProgress)
360  {
361  /* refresh bs timer */
362  //link->send_timer_bs = isotp_user_get_ms() + ISO_TP_DEFAULT_RESPONSE_TIMEOUT;
363  switch (ptr2FlowControlFrame->as.flow_control.FS)
364  {
365  case ( IsoTP_FlowStatus::OVERFLOW ):
366  {
367  /* overflow */
368  objectHandle.FlowControl.ProtocolStatus = IsoTP_RxTxStatus::IsoTP_SendError;
369  objectHandle.FlowControl.ProtocolResult = IsoTP_ProtocolResultType::RESULT_BUFFER_OVFLW;
370  returnValue = E_OK;
371  break;
372  }
373  case ( IsoTP_FlowStatus::WAIT ):
374  {
375  /* wait */
376  objectHandle.FlowControl.WaitForTransmitCounter += 1;
377  /* wait exceed allowed count */
378  //if ( objectHandle.waitForTransmitCounter > objectConfig.ptr2Config->maxWaitForTransmitCylcles )

382  objectHandle.FlowControl.ProtocolResult = IsoTP_ProtocolResultType::RESULT_WFT_OVRN;
383  }
384  returnValue = E_OK;
385  break;
386  }
387  case ( IsoTP_FlowStatus::CONTINUE ):
388  {
389  /* permit send */
390  if ( ptr2FlowControlFrame->as.flow_control.BS == 0)
391  {
392  objectHandle.FlowControl.RemainingBlockSize = ISOTP_INVALID_BS;

401  break;
402  }
403  default:
404 #if ( ISOTP_DEV_ERROR_DETECT == STD_ON )
405  // should not happen
406  ISOTP_DET_REPORTERROR(ISOTP_E_UNDEFINED, TM_ISOTP_RECEIVE_FLOWCONTROL_ID);

409  }

582  /* send data if bs_remain is invalid or bs_remain large than zero */
583  if ( ( ISOTP_INVALID_BS == objectHandle.FlowControl.RemainingBlockSize || objectHandle.FlowControl.RemainingBlockSize > 0) )
584  {
585  /* and if st_min is zero or go beyond interval time */
586  if ( objectHandle.FlowControl.MinSeparationTime == 0 ||
587  ( (objectHandle.FlowControl.MinSeparationTime != 0) && (IsoTpTimeAfter(GetMillisecond(), objectHandle.FlowControl.stTimer) ) )
588  )
589  {
590  if ( SendConsecutiveFrame() == E_OK )
591  {
592  if ( ISOTP_INVALID_BS != objectHandle.FlowControl.RemainingBlockSize )
593  {
594  objectHandle.FlowControl.RemainingBlockSize -= 1;
595  }
596  objectHandle.FlowControl.bsTimer = GetMillisecond() + objectConfig.ptr2Config->maxResponseTimeout;
597  objectHandle.FlowControl.stTimer = GetMillisecond() + objectHandle.FlowControl.MinSeparationTime;
598  /* check if send finish */

602  objectHandle.FlowControl.ProtocolResult = IsoTP_ProtocolResultType::RESULT_OK;
603  }
604  }
605  else
606  {
607  objectHandle.FlowControl.ProtocolStatus = IsoTP_RxTxStatus::IsoTP_SendError;
608  }
609  } // end of test the separation time
610  } // end of test the block size
611 
612  /* check timeout */
613  if (IsoTpTimeAfter(GetMillisecond(), objectHandle.FlowControl.bsTimer))
614  {
615  objectHandle.FlowControl.ProtocolStatus = IsoTP_RxTxStatus::IsoTP_SendError;
616  objectHandle.FlowControl.ProtocolResult = IsoTP_ProtocolResultType::RESULT_TIMEOUT_BS;
617  }
618  }
619 
620  if ( objectHandle.FlowControl.ProtocolStatus == IsoTP_RxTxStatus::IsoTP_ReceiveInProgress )
621  {
622  /* check timeout */
623  if (IsoTpTimeAfter(GetMillisecond(), objectHandle.FlowControl.bsTimer))
624  {
625  objectHandle.FlowControl.ProtocolStatus = IsoTP_RxTxStatus::IsoTP_ReceiveIdle;
626  objectHandle.FlowControl.ProtocolResult = IsoTP_ProtocolResultType::RESULT_TIMEOUT_CR;
627  }
628 
629  }
630 
631 
632  returnValue = E_OK;
633  }
634 return returnValue;

638 FUNC(Std_ReturnType, TM_ISOTP_CODE) IsoTp::RxIndication(
639  CONST(ComStack_PduType, AUTOMATIC) rxPduId,
640  P2VAR(ComStack_CanMessageType, AUTOMATIC, AUTOMATIC) ptr2Sdu)
641 {
642  Std_ReturnType returnValue = E_NOT_OK;
643 
644  if ( GetState() != IsoTP_StatesType::IsoTP_ON )
645  {
646 #if ( ISOTP_DEV_ERROR_DETECT == STD_ON )
647  ISOTP_DET_REPORTERROR(ISOTP_E_UNINIT, TM_ISOTP_RXINDICATION_ID);
648 #endif // ( ISOTP_DEV_ERROR_DETECT == STD_ON )
649  }
650  else
651  {
652  if ( ( ptr2Sdu != NULL_PTR ) &&
653  (rxPduId < IsoTP_Rx_unknownPdu ) )
654  {
655  if ( ( ptr2Sdu->CanMsgHeader.DLC <2 ) ||
656  ( ptr2Sdu->CanMsgHeader.DLC >8 ) )
657  {
658  //returnValue = NON_STD_RET_VALUE(IsoTp_ReturnErrorType::RET_E_LENGTH);
659  returnValue = E_NOT_OK;
660  }
661  else
662  {
663 #if ( ISOTP_DECODE_LEVEL == DECODE_ON_INTERRUPT )
664  CanIsoTP_8ByteData * msg = (CanIsoTP_8ByteData *)ptr2Sdu->ptr2Data;
665  switch (msg->as.common.type)
666  {
667 /* ####### Flow Control Frame ##### */
668  case (IsoTP_FrameType::IsoTP_FlowControl):
669  {
670  if ( ptr2Sdu->CanMsgHeader.DLC >= 3 )
671  {
672  returnValue = ReceiveFlowControl((CanIsoTP_8ByteData*)ptr2Sdu->ptr2Data);
673  // may be do error handling on development
674 #if ( ISOTP_DEV_ERROR_DETECT == STD_ON )
675  if ( returnValue != E_OK ) ISOTP_DET_REPORTERROR(ISOTP_E_INVALID_FRAME, TM_ISOTP_RXINDICATION_ID);
676 #else
677  returnValue = E_OK; // report ok to clear the buffer
678 #endif // ( ISOTP_DEV_ERROR_DETECT == STD_ON )
679  }
680  else
681  {
682  // may be do error handling on development
683 #if ( ISOTP_DEV_ERROR_DETECT == STD_ON )
684  ISOTP_DET_REPORTERROR(ISOTP_E_INVALID_FRAMELENGTH, TM_ISOTP_RXINDICATION_ID);
685 #else
686  returnValue = E_OK; // report ok to clear the buffer
687 #endif // ( ISOTP_DEV_ERROR_DETECT == STD_ON )
688  }
689  break;
690  }
691 /* ####### Single Frame ##### */
692  case (IsoTP_FrameType::IsoTP_SingleFrame):
693  {
694  returnValue = ReceiveSingleFrame((CanIsoTP_8ByteData*)ptr2Sdu->ptr2Data);
695  if ( returnValue == E_OK )
696  {
697  objectHandle.FlowControl.ProtocolStatus = IsoTP_RxTxStatus::IsoTP_ReceiveFull;
698  }
699  break;
700  }
701 /* ####### First Segment Frame ##### */
702  case (IsoTP_FrameType::IsoTP_FirstFrame):
703  {
704  returnValue = ReceiveFirstFrame((CanIsoTP_8ByteData*)ptr2Sdu->ptr2Data);
705  if ( returnValue == E_OK )
706  {
709  SendFlowControl(IsoTP_FlowStatus::CONTINUE);
710  }
711  else if ( returnValue == NON_STD_RET_VALUE(IsoTp_ReturnErrorType::RET_E_OVERFLOW) )
712  {
715  SendFlowControl(IsoTP_FlowStatus::OVERFLOW);
716  returnValue = E_OK; // report ok to clear the buffer
717  }
718  break;
719  }
720 /* ####### Consecutive Segment Frame ##### */
721  case (IsoTP_FrameType::IsoTP_ConsecutiveFrame):
722  {
723  returnValue = ReceiveConsecutiveFrame((CanIsoTP_8ByteData*)ptr2Sdu->ptr2Data, ptr2Sdu->CanMsgHeader.DLC - 1);
724  if ( returnValue == NON_STD_RET_VALUE(IsoTp_ReturnErrorType::RET_LAST_OK) )
725  {
728  SendFlowControl(IsoTP_FlowStatus::CONTINUE);
729  }
730  // may be do error handling on development
731 #if ( ISOTP_DEV_ERROR_DETECT == STD_ON )
732  if ( returnValue != E_OK ) ISOTP_DET_REPORTERROR(ISOTP_E_INVALID_FRAME, TM_ISOTP_RXINDICATION_ID);
733 #else
734  returnValue = E_OK; // report ok to clear the buffer
735 #endif // ( ISOTP_DEV_ERROR_DETECT == STD_ON )
736  break;
737  }
738  default:
739 /* ####### Unknown Frame ##### */
740 #if ( ISOTP_DEV_ERROR_DETECT == STD_ON )
741  // should not happen
742  ISOTP_DET_REPORTERROR(ISOTP_E_UNDEFINED, TM_ISOTP_RXINDICATION_ID);
743 #endif // ( ISOTP_DEV_ERROR_DETECT == STD_ON )
744  break; // nothing to do
745 
746  }
747 /* ####### End Of Switch Frame Type ##### */
748  }
749 #else
750 
751 #endif // ( ISOTP_DECODE_LEVEL == DECODE_ON_INTERRUPT )
752  }
753  }
754  return returnValue;
755 }
