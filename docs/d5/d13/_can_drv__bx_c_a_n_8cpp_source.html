233 FUNC(void, TM_CAN_CODE) CanDrv_bxCAN::TxTask(void)
234 {
235  uint32_t txBufferpos = 0;
236 
237  if ( Can::GetDrvState() == CanDrvStates::Ready )
238  {
239  for (txBufferpos = 0; txBufferpos < CAN_TX_SW_BUFFER_SIZE; txBufferpos++ )
240  {
242  if ( HAL_CAN_GetTxMailboxesFreeLevel(&bxCanHdl[CanMasterController]) > 0 )
243  {
244  switch ( canTxBuffer[txBufferpos].State)
245  {
246  case bxCan_TxMessageStateType::TxMessage_rdy2tx:
247  // we block for transmit
248  canTxBuffer[txBufferpos].State = bxCan_TxMessageStateType::TxMessage_transmit;
249 
250  // give the message via HAL to the CAN Controller
251  if ( HAL_CAN_AddTxMessage(&bxCanHdl[CanMasterController], &canTxBuffer[txBufferpos].Header, canTxBuffer[txBufferpos].Data, &canTxBuffer[txBufferpos].Mailbox) != HAL_OK)
252  {
253  canTxBuffer[txBufferpos].State = bxCan_TxMessageStateType::TxMessage_txerr;
254  Error_Handler();
255  }
256  break;
257  case bxCan_TxMessageStateType::TxMessage_transmit:
258  if ( !(HAL_CAN_IsTxMessagePending(&bxCanHdl[CanMasterController], canTxBuffer[txBufferpos].Mailbox)) )
259  {
260  canTxBuffer[txBufferpos].State = bxCan_TxMessageStateType::TxMessage_txok;
261  }
262  break;
263  case bxCan_TxMessageStateType::TxMessage_txok:
264  if ( 1 ) // maybe we want to wait until clear up the mailbox
265  {
266  canTxBuffer[txBufferpos].Mailbox = 0;
267  canTxBuffer[txBufferpos].Header = {0};
268  canTxBuffer[txBufferpos].State = bxCan_TxMessageStateType::TxMessage_free;
269  }
270  break;
271  default:

281 FUNC(Std_ReturnType, TM_CAN_CODE) CanDrv_bxCAN::FindFreeFilter(
282  P2VAR(uint8_t, AUTOMATIC, AUTOMATIC) ptr2FilterBankNumber,
283  P2VAR(uint8_t, AUTOMATIC, AUTOMATIC) ptr2FilterNumber,
284  P2VAR(uint8_t, AUTOMATIC, AUTOMATIC) ptr2FilterMode)
285 {
286  P2VAR(CAN_TypeDef, AUTOMATIC, AUTOMATIC) ptr2Instance = bxCanHdl[CanMasterController].Instance;
287  Std_ReturnType returnValue = E_NOT_OK;
288  uint32_t filterBankRegValue = ptr2Instance->FA1R;
289  uint32_t filterModeRegValue = ptr2Instance->FM1R;
290 
291  /* the STM HAL library has no function to find a free hardware filter
292  * we test only for active filter */
293  uint8_t testBank = 0;
294  uint8_t foundBank = FALSE;
295  uint8_t filterCounter = 0;
296  uint8_t filterIncrement = 0;
297 
299  while ((filterCounter < CAN1_USED_RX_HW_FILTER) && (foundBank == FALSE ))
300  {
301  foundBank = 0x1 & ~(filterBankRegValue>>testBank);
302  if ( (0x1 & (filterModeRegValue>>testBank)) == CAN_FILTERMODE_IDLIST)
303  {
304  filterIncrement = 2;
305  // second filter inside a filter bank used as ID filter is free
306  if ( (*ptr2FilterMode == CAN_FILTERMODE_IDLIST ) &&
307  (ptr2Instance->sFilterRegister[testBank].FR2 == 0xFFFFFFFF) )
308  {
309  foundBank = TRUE;
310  filterCounter++; // single step when select a second filter
313  *ptr2FilterMode |= 0x80;
314  }
315 
316  }
317  else
318  {
319  filterIncrement = 1;
320  }
321 
322  if (foundBank == FALSE )
323  {
324  testBank++;
325  filterCounter += filterIncrement;
326  }
327  else /* we have found a free filter */
328  {
329  *ptr2FilterBankNumber = testBank;
330  *ptr2FilterNumber = filterCounter;
331  returnValue = E_OK;
332  }
333  }
334 
335  return returnValue;
336 }
